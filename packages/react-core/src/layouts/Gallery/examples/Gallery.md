```js
  <Gallery>

    <GalleryItem>Gallery Item</GalleryItem>

    <GalleryItem>Gallery Item</GalleryItem>
    <GalleryItem>Gallery Item</GalleryItem>
    <GalleryItem>Gallery Item</GalleryItem>
  </Gallery>
);
```
